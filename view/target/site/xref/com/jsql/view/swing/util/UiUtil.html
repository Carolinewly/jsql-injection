228     private static void loadFonts() {
229         
230         var ge = GraphicsEnvironment.getLocalGraphicsEnvironment();
231         
232         try (InputStream fontStream = new BufferedInputStream(Objects.requireNonNull(UiUtil.class.getClassLoader().getResourceAsStream("swing/font/UbuntuMono-R-ctrlchar.ttf")))) {
233             
234             var ubuntuFont = Font.createFont(Font.TRUETYPE_FONT, fontStream);
235             ge.registerFont(ubuntuFont);
236             
237         } catch (FontFormatException | IOException e) {
238             
239             LOGGER.log(LogLevelUtil.CONSOLE_ERROR, "Loading Font Ubuntu Mono with control characters failed", e);
240         }
241     }

422         List<Image> images = new ArrayList<>();
423         
424         // Fix #2154: NoClassDefFoundError on read()

444     public static void drawPlaceholder(JTextComponent textComponent, Graphics g, String placeholderText, int y) {
445         
446         int w = textComponent.getWidth();
447         
448         ((Graphics2D) g).setRenderingHint(RenderingHints.KEY_TEXT_ANTIALIASING, RenderingHints.VALUE_TEXT_ANTIALIAS_ON);
449         
450         var ins = textComponent.getInsets();
451         var fm = g.getFontMetrics();
452         
453         int c0 = textComponent.getBackground().getRGB();
454         int c1 = textComponent.getForeground().getRGB();
455         var m = 0xfefefefe;
456         int c2 = ((c0 & m) >>> 1) + ((c1 & m) >>> 1);
457         
458         g.setColor(new Color(c2, true));
459         g.setFont(textComponent.getFont().deriveFont(Font.ITALIC));
460 
461         g.drawString(
462             placeholderText,
463             ComponentOrientation.RIGHT_TO_LEFT.equals(textComponent.getComponentOrientation())
464             ? w - (fm.stringWidth(placeholderText) + ins.left + 2)
465             : ins.left + 2,
466             y
467         );
468     }

483 
484             @Override

486
