11  package com.jsql.view.swing.tab;
12  
13  import com.jsql.util.I18nUtil;
14  import com.jsql.view.swing.text.JToolTipI18n;
15  import com.jsql.view.swing.ui.CustomMetalTabbedPaneUI;
16  import com.jsql.view.swing.util.I18nViewUtil;
17  import com.jsql.view.swing.util.MediatorHelper;
18  import com.jsql.view.swing.util.UiUtil;
19  
20  import javax.swing.*;
21  import java.awt.*;
22  import java.awt.event.MouseAdapter;
23  import java.awt.event.MouseEvent;
24  
25  /**
26   * Panel on the left with functionalities like webshell, file reading and admin page finder.
27   */
28  public class TabManagers extends TabbedPaneWheeled {
29      
30      /**
31       * Create manager panel.
32       */
33      public TabManagers() {
34          
35          this.setName("tabManagers");
36          
37          this.setUI(new CustomMetalTabbedPaneUI() {
38              
39              @Override
40              protected int calculateTabWidth(int tabPlacement, int tabIndex, FontMetrics metrics) {
41                  
42                  return Math.max(75, super.calculateTabWidth(tabPlacement, tabIndex, metrics));
43              }
44          });
45  
46          this.addChangeListener(e -> MediatorHelper.tabManagers().setSelectedIndex(TabManagers.this.getSelectedIndex()));
47          
48          this.setMaximumSize(new Dimension(this.getMaximumSize().width, 25));
49          this.setPreferredSize(new Dimension(this.getPreferredSize().width, 25));
50          
51          this.addMouseClickMenu();
52          
53          this.setMinimumSize(new Dimension(100, 0));
54          
55          this.buildI18nTab("DATABASE_TAB", "DATABASE_TOOLTIP", UiUtil.ICON_DATABASE_SERVER, 0);
56          this.buildI18nTab("ADMINPAGE_TAB", "ADMINPAGE_TOOLTIP", UiUtil.ICON_ADMIN_SERVER, 1);
57          this.buildI18nTab("FILE_TAB", "FILE_TOOLTIP", UiUtil.ICON_FILE_SERVER, 2);
58          this.buildI18nTab("WEBSHELL_TAB", "WEBSHELL_TOOLTIP", UiUtil.ICON_SHELL_SERVER, 3);
59          this.buildI18nTab("SQLSHELL_TAB", "SQLSHELL_TOOLTIP", UiUtil.ICON_SHELL_SERVER, 4);
60          this.buildI18nTab("UPLOAD_TAB", "UPLOAD_TOOLTIP", UiUtil.ICON_UPLOAD, 5);
61          this.buildI18nTab("BRUTEFORCE_TAB", "BRUTEFORCE_TOOLTIP", UiUtil.ICON_BRUTER, 6);
62          this.buildI18nTab("CODER_TAB", "CODER_TOOLTIP", UiUtil.ICON_CODER, 7);
63          this.buildI18nTab("SCANLIST_TAB", "SCANLIST_TOOLTIP", UiUtil.ICON_SCANLIST, 8);
64      }
65      
66      private void buildI18nTab(String keyLabel, String keyTooltip, Icon icon, int index) {
67          
68          final var refTooltip = new JToolTipI18n[]{ new JToolTipI18n(I18nUtil.valueByKey(keyTooltip)) };
69          
70          JLabel labelTab = new JLabel(I18nUtil.valueByKey(keyLabel), icon, SwingConstants.CENTER) {
71              
72              @Override
73              public JToolTip createToolTip() {
74                  
75                  JToolTipI18n tipI18n = new JToolTipI18n(I18nUtil.valueByKey(keyTooltip));
76                  refTooltip[0] = tipI18n;
77                  
78                  return tipI18n;
79              }
80          };
81          
82          labelTab.addMouseListener(new MouseAdapter() {
83              
84              @Override
85              public void mousePressed(MouseEvent e) {
86                  
87                  MediatorHelper.tabManagers().setSelectedIndex(index);
88                  TabManagers.this.setSelectedIndex(index);
89                  super.mousePressed(e);
90              }
91          });
92  
93          // Required for i18n to work
94          this.addTab(I18nUtil.valueByKey(keyLabel), icon, null);
95          this.setTabComponentAt(
96              this.indexOfTab(I18nUtil.valueByKey(keyLabel)),
97              labelTab
98          );
99  
100         I18nViewUtil.addComponentForKey(keyLabel, labelTab);
101         I18nViewUtil.addComponentForKey(keyTooltip, refTooltip[0]);
102         
103         labelTab.setToolTipText(I18nUtil.valueByKey(keyTooltip));
104         labelTab.addMouseListener(new TabMouseAdapter(this));
105     }
106 }
