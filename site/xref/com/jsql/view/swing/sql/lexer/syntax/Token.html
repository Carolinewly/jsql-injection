18  package com.jsql.view.swing.sql.lexer.syntax;
19  
20  /**
21   * A generic token class.
22   */
23  public abstract class Token {
24  
25      /**
26       * The state of the tokenizer is undefined.
27       */
28      public static final int UNDEFINED_STATE = -1;
29  
30      /**
31       * The initial state of the tokenizer.
32       * Anytime the tokenizer returns to this state,
33       * the tokenizer could be restarted from that point
34       * with side effects.
35       */
36      public static final int INITIAL_STATE = 0;
37  
38      /**
39       * A unique ID for this type of token.
40       * Typically, ID numbers for each type will
41       * be static variables of the Token class.
42       * 
43       * @return an ID for this type of token.
44       */
45      public abstract int getID();
46      
47      /**
48       * A description of this token.  The description should
49       * be appropriate for syntax highlighting.  For example
50       * "comment" might be returned for a comment.  This should
51       * make it easy to do html syntax highlighting.  Just use
52       * style sheets to define classes with the same name as
53       * the description and write the token in the html file
54       * with that css class name.
55       *
56       * @return a description of this token.
57       */
58      public abstract String getDescription();
59      
60      /**
61       * The actual meat of the token.
62       *
63       * @return a string representing the text of the token.
64       */
65      public abstract String getContents();
66      
67      /**
68       * Determine if this token is a comment.  Sometimes comments should be
69       * ignored (compiling code) other times they should be used
70       * (syntax highlighting).  This provides a method to check
71       * in case you feel you should ignore comments.
72       *
73       * @return true if this token represents a comment.
74       */
75      public abstract boolean isComment();
76      
77      /**
78       * Determine if this token is whitespace.  Sometimes whitespace should be
79       * ignored (compiling code) other times they should be used
80       * (code beautification).  This provides a method to check
81       * in case you feel you should ignore whitespace.
82       *
83       * @return true if this token represents whitespace.
84       */
85      public abstract boolean isWhiteSpace();
86      
87      /**
88       * Determine if this token is an error.  Lets face it, not all code
89       * conforms to spec. The lexer might know about an error
90       * if a string literal is not closed, for example.
91       *
92       * @return true if this token is an error.
93       */
94        public abstract boolean isError();
95      
96      /**
97       * get the line number of the input on which this token started
98       * 
99       * @return the line number of the input on which this token started
100      */
101     public abstract int getLineNumber();
102 
103     /**
104      * get the offset into the input in characters at which this token started
105      *
106      * @return the offset into the input in characters at which this token started
107      */
108     public abstract int getCharBegin();
109 
110     /**
111      * get the offset into the input in characters at which this token ended
112      *
113      * @return the offset into the input in characters at which this token ended
114      */
115     public abstract int getCharEnd();
116     
117     /**
118      * get a String that explains the error, if this token is an error.
119      * 
120      * @return a  String that explains the error, if this token is an error, null otherwise.
121      */
122     public abstract String errorString();
123 
124     /**
125      * Get an integer representing the state the tokenizer is in after
126      * returning this token.
127      * Those who are interested in incremental tokenizing for performance
128      * reasons will want to use this method to figure out where the tokenizer
129      * may be restarted.  The tokenizer starts in Token.INITIAL_STATE, so
130      * any time that it reports that it has returned to this state, the
131      * tokenizer may be restarted from there.
132      */
133     public abstract int getState();
134 }
