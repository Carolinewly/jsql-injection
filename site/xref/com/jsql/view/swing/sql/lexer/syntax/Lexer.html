18  package com.jsql.view.swing.sql.lexer.syntax;
19  
20  import java.io.IOException;
21  
22  /**
23   * A lexer should implement this interface.
24   */
25  public interface Lexer {
26      
27      /**
28       * Returns the next token.
29       *
30       * @return the next token
31       */
32      Token getNextToken() throws IOException;
33  
34      /**
35       * Closes the current input stream, and resets the scanner to read from a new input stream.
36       * All internal variables are reset, the old input stream  cannot be reused
37       * (content of the internal buffer is discarded and lost).
38       * The lexical state is set to the initial state.
39       * Subsequent tokens read from the lexer will start with the line, char, and column
40       * values given here.
41       *
42       * @param reader The new input.
43       * @param yyline The line number of the first token.
44       * @param yychar The position (relative to the start of the stream) of the first token.
45       * @param yycolumn The position (relative to the line) of the first token.
46       */
47      void reset(java.io.Reader reader, int yyline, int yychar, int yycolumn);
48  }
