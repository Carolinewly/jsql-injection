16  /**
17   * A scrollpane like component, where the scroll bars are floating over the
18   * scrollable view to indicate the current scroll positions.
19   * The scroll indicators appear smoothly during scroll events and disappear
20   * smoothly afterwards.
21   * <p>
22   * The scrollbars can be dragged just as normal.</p>
23   * <p>
24   * The usage is similar to a classic scrollpane.</p>
25   *
26   * @author Jolly Littlebottom
27   */
28  public class JScrollIndicator extends JLayeredPane {
29      
30      /**
31       * Log4j logger sent to view.
32       */
33      private static final Logger LOGGER = LogManager.getRootLogger();
34  
35      private static final int SCROLL_BAR_ALPHA_ROLLOVER = 100;
36      private static final int SCROLL_BAR_ALPHA = 25;
37  
38      private static final Color THUMB_COLOR = Color.DARK_GRAY;
39      private static final int THUMB_THICKNESS = 15;
40      private static final int THUMB_MIN_SIZE = 48;
41      private static final int THUMB_MARGIN = 0;
42  
43      private final JScrollPane scrollPane;
44      private final ControlPanel controlPanel;
45  
46      /**
47       * Creates a <code>JScrollIndicator</code> that displays the contents of the
48       * specified component, where both horizontal and vertical scrollbars appear
49       * whenever the component's contents are larger than the view and scrolling
50       * in underway or the mouse is over the scrollbar position.
51       *
52       * see #setViewportView
53       * @param view the component to display in the scrollpane's viewport
54       */
55      public JScrollIndicator(final JComponent view) {
56          
57          this(view, ScrollPaneConstants.VERTICAL_SCROLLBAR_AS_NEEDED, ScrollPaneConstants.HORIZONTAL_SCROLLBAR_AS_NEEDED);
58      }
59      
60      public JScrollIndicator(final JComponent view, int scrollPaneConstants) {
61          
62          this(view, ScrollPaneConstants.VERTICAL_SCROLLBAR_AS_NEEDED, scrollPaneConstants);
63      }
64  
65      /**
66       * Creates a <code>JScrollIndicator</code> that displays the view component
67       * in a viewport whose view position can be controlled with a pair of
68       * scrollbars.
69       * The scrollbar policies specify when the scrollbars are displayed,
70       * For example, if <code>vsbPolicy</code> is
71       * <code>JScrollPane.VERTICAL_SCROLLBAR_AS_NEEDED</code>
72       * then the vertical scrollbar only appears if the view doesn't fit
73       * vertically. The available policy settings are listed at
74       * {link #JScrollPane.setVerticalScrollBarPolicy} and
75       * {link #JScrollPane.setHorizontalScrollBarPolicy}.
76       *
77       * @param view the component to display in the scrollpanes viewport
78       * @param vsbPolicy an integer that specifies the vertical scrollbar policy
79       * @param hsbPolicy an integer that specifies the horizontal scrollbar policy
80       */
81      public JScrollIndicator(final JComponent view, int vsbPolicy, int hsbPolicy) {
82          
83          this.scrollPane = new JScrollPane(view, vsbPolicy, hsbPolicy);
84          this.scrollPane.setBorder(BorderFactory.createEmptyBorder());
85          this.add(this.scrollPane, JLayeredPane.DEFAULT_LAYER);
86  
87          this.controlPanel = new ControlPanel(this.scrollPane);
88          this.add(this.controlPanel, JLayeredPane.PALETTE_LAYER);
89  
90          this.addComponentListener(
91              new ComponentAdapter() {
92                  
93                  @Override
94                  public void componentResized(ComponentEvent e) {
95                      
96                      // listen to changes of JLayeredPane size
97                      JScrollIndicator.this.scrollPane.setSize(JScrollIndicator.this.getSize());
98                      JScrollIndicator.this.scrollPane.getViewport().revalidate();
99                      JScrollIndicator.this.controlPanel.setSize(JScrollIndicator.this.getSize());
100                     JScrollIndicator.this.controlPanel.revalidate();
101                 }
102             }
103         );
104     }
105 
106     /**
107      * Returns the scroll pane used by this scroll indicator.
108      * Use carefully (e.g. to set unit increments) because not all changes have an
109      * effect. You have to write listeners in this cases (e.g. for changing the
110      * scrollbar policy)
111      * 
112      * @return
113      */
114     public JScrollPane getScrollPane() {
115         return this.scrollPane;
116     }

118     private class ControlPanel extends JPanel {
119 
120         private ControlPanel(JScrollPane scrollPane) {
121             
122             this.setLayout(new BorderLayout());
123             this.setOpaque(false);
124 
125             JMyScrollBar vScrollBar = new JMyScrollBar(Adjustable.VERTICAL);
126             scrollPane.setVerticalScrollBar(vScrollBar);
127             scrollPane.remove(vScrollBar);
128             
129             if (scrollPane.getVerticalScrollBarPolicy() != ScrollPaneConstants.VERTICAL_SCROLLBAR_NEVER) {
130                 
131                 this.add(vScrollBar, BorderLayout.EAST);
132             }
133 
134             JMyScrollBar hScrollBar = new JMyScrollBar(Adjustable.HORIZONTAL);
135             scrollPane.setHorizontalScrollBar(hScrollBar);
136             scrollPane.remove(hScrollBar);
137             
138             if (scrollPane.getHorizontalScrollBarPolicy() != ScrollPaneConstants.HORIZONTAL_SCROLLBAR_NEVER) {
139                 
140                 this.add(hScrollBar, BorderLayout.SOUTH);
141             }
142         }
143     }

145     private class JMyScrollBar extends JScrollBar {
146 
147         protected final transient MyScrollBarUI scrollUI;
148 
149         public JMyScrollBar(int direction) {
150             
151             super(direction);
152 
153             this.scrollUI = new MyScrollBarUI(this);
154             super.setUI(this.scrollUI);
155             this.setUnitIncrement(64);
156             int size = THUMB_THICKNESS + THUMB_MARGIN;
157             this.setPreferredSize(new Dimension(size, size));
158             this.scrollUI.setVisible();
159             
160             this.addMouseListener(new MouseAdapter() {
161                 
162                 @Override
163                 public void mouseEntered(MouseEvent e) {
164                     
165                     JMyScrollBar.this.scrollUI.setVisible();
166                 }
167 
168                 @Override
169                 public void mouseExited(MouseEvent e) {
170                     
171                     JMyScrollBar.this.scrollUI.setVisible();
172                 }
173             });
174 
175             this.addAdjustmentListener(adjustmentEvent -> this.scrollUI.setVisible());
176         }
177 
178         @Override
179         public void setUI(ScrollBarUI ui) {
180             // Nothing
181         }
182 
183         @Override
184         public void updateUI() {
185             // Nothing
186         }
187 
188         @Override
189         public void paint(Graphics g) {
190             this.scrollUI.paintThumb(g); // just the thumb
191         }
192 
193         @Override
194         public void repaint(Rectangle r) {
195             
196             JScrollIndicator scrollIndicator = JScrollIndicator.this;
197             
198             // Fix #15956: NullPointerException on convertRectangle()
199             try {
200                 var rect = SwingUtilities.convertRectangle(this, r, scrollIndicator);
201                 rect.grow(1, 1);
202                 // ensure for a translucent thumb, that the view is first painted
203                 scrollIndicator.repaint(rect);
204                 
205             } catch (NullPointerException e) {
206                 
207                 LOGGER.log(LogLevelUtil.CONSOLE_JAVA, e, e);
208             }
209         }
210     }

212     public class MyScrollBarUI extends BasicScrollBarUI {
213         
214         private final JMyScrollBar myScrollBar;
215         private int alpha = 0;
216 
217         private MyScrollBarUI(JMyScrollBar scrollBar) {
218             
219             this.myScrollBar = scrollBar;
220         }
221 
222         @Override
223         protected void installComponents() {
224             
225             this.incrButton = new JButton();
226             this.decrButton = new JButton();
227             
228             if (this.myScrollBar.getOrientation() == Adjustable.HORIZONTAL) {
229                 
230                 int size = THUMB_THICKNESS + THUMB_MARGIN; // let lower right corner empty
231                 this.incrButton.setPreferredSize(new Dimension(size, size));
232                 
233             } else {
234                 
235                 this.incrButton.setPreferredSize(new Dimension(THUMB_MARGIN, THUMB_MARGIN));
236             }
237             
238             this.decrButton.setPreferredSize(new Dimension(THUMB_MARGIN, THUMB_MARGIN));
239         }
240 
241         @Override
242         protected void installDefaults() {
243             
244             super.installDefaults();
245 
246             // ensure the minimum size of the thumb
247             int w = this.minimumThumbSize.width;
248             int h = this.minimumThumbSize.height;
249             
250             if (this.myScrollBar.getOrientation() == Adjustable.VERTICAL) {
251                 
252                 h = Math.max(h, Math.min(this.maximumThumbSize.height, THUMB_MIN_SIZE));
253                 
254             } else {
255                 
256                 w = Math.max(w, Math.min(this.maximumThumbSize.width, THUMB_MIN_SIZE));
257             }
258             
259             this.minimumThumbSize = new Dimension(w, h);
260         }
261 
262         private void paintThumb(Graphics g) {
263             
264             int alphaThumb =
265                 this.isThumbRollover()
266                 ? SCROLL_BAR_ALPHA_ROLLOVER
267                 : SCROLL_BAR_ALPHA;
268 
269             g.setColor(
270                 new Color(
271                     this.getAlphaColor().getRed(),
272                     this.getAlphaColor().getGreen(),
273                     this.getAlphaColor().getBlue(),
274                     alphaThumb
275                 )
276             );
277             
278             Rectangle thumbBounds = this.getThumbBounds();
279 
280             int x = thumbBounds.x;
281             int y = thumbBounds.y;
282             int w = thumbBounds.width;
283             int h = thumbBounds.height;
284 
285             if (this.myScrollBar.getOrientation() == Adjustable.VERTICAL) {
286                 
287                 w -= THUMB_MARGIN;
288                 
289             } else {
290                 
291                 h -= THUMB_MARGIN;
292             }
293 
294             g.fillRect(x, y, w, h);
295         }
296 
297         private Color getAlphaColor() {
298             
299             if (this.alpha == 100) {
300                 
301                 return JScrollIndicator.THUMB_COLOR;
302             }
303             
304             int rgb = JScrollIndicator.THUMB_COLOR.getRGB() & 0xFFFFFF; // color without alpha values
305             rgb |= (this.alpha / 100 * 255) << 24; // add alpha value
306             
307             return new Color(rgb, true);
308         }
309 
310         public void setAlpha(int alpha) {
311             
312             this.alpha = alpha;
313             this.myScrollBar.repaint(this.getThumbBounds());
314         }
315 
316         public void setVisible() {
317             
318             this.myScrollBar.repaint(this.getThumbBounds());
319         }
320     }
