42      public void reloadNodes() {
43          
44          // I18n of tree empty node
45          if (this.isRootVisible()) {
46              
47              DefaultTreeModel model = (DefaultTreeModel) this.getModel();
48              DefaultMutableTreeNode root = (DefaultMutableTreeNode) model.getRoot();
49              model.reload(root);
50              this.revalidate();
51          }
52      }
53  
54      public void reset() {
55          
56          // Tree model for refreshing the tree
57          DefaultTreeModel treeModel = (DefaultTreeModel) this.getModel();
58          // The tree root
59          DefaultMutableTreeNode root = (DefaultMutableTreeNode) treeModel.getRoot();
60  
61          // Remove tree nodes
62          root.removeAllChildren();
63          // Refresh the root
64          treeModel.nodeChanged(root);
65          // Refresh the tree
66          treeModel.reload();
67          
68          this.setRootVisible(true);
69      }

71      public void addColumns(List<Column> columns) {
72          
73          // Tree model, update the tree (refresh, add node, etc)
74          DefaultTreeModel treeModel = (DefaultTreeModel) this.getModel();
75  
76          // The table to update
77          DefaultMutableTreeNode tableNode = null;
78  
79          // Loop into the list of columns
80          for (Column column: columns) {
81              
82              // Create a node model with the column element
83              AbstractNodeModel newTreeNodeModel = new NodeModelColumn(column);
84  
85              // Create the node
86              var newNode = new DefaultMutableTreeNode(newTreeNodeModel);
87              
88              // Get the parent table
89              tableNode = this.getTreeNodeModels().get(column.getParent());
90              
91              // Fix #1805 : NullPointerException on tableNode.getChildCount()
92              if (tableNode != null) {
93                  
94                  // Add the column to the table
95                  treeModel.insertNodeInto(newNode, tableNode, tableNode.getChildCount());
96              }
97          }
98  
99          if (tableNode != null) {
100             
101             // Open the table node
102             this.expandPath(new TreePath(tableNode.getPath()));
103             
104             // The table has just been search (avoid double check)
105             ((AbstractNodeModel) tableNode.getUserObject()).setLoaded(true);
106         }
107     }

109     public void addDatabases(List<Database> databases) {
110         
111         // Tree model, update the tree (refresh, add node, etc)
112         DefaultTreeModel treeModel = (DefaultTreeModel) this.getModel();
113 
114         // First node in tree
115         DefaultMutableTreeNode root = (DefaultMutableTreeNode) treeModel.getRoot();
116 
117         // Loop into the list of databases
118         for (Database database: databases) {
119             
120             // Create a node model with the database element
121             AbstractNodeModel newTreeNodeModel = new NodeModelDatabase(database);
122             // Create the node
123             var newNode = new DefaultMutableTreeNode(newTreeNodeModel);
124             // Save the node
125             this.getTreeNodeModels().put(database, newNode);
126             // Add the node to the tree
127             root.add(newNode);
128         }
129 
130         // Refresh the tree
131         treeModel.reload(root);
132         
133         // Open the root node
134         this.expandPath(new TreePath(root.getPath()));
135         this.setRootVisible(false);
136     }

138     public void addTables(List<Table> tables) {
139         
140         // Tree model, update the tree (refresh, add node, etc)
141         DefaultTreeModel treeModel = (DefaultTreeModel) this.getModel();
142 
143         // The database to update
144         DefaultMutableTreeNode databaseNode = null;
145 
146         // Loop into the list of tables
147         for (Table table: tables) {
148             
149             // Create a node model with the table element
150             AbstractNodeModel newTreeNodeModel = new NodeModelTable(table);
151             
152             // Create the node
153             var newNode = new DefaultMutableTreeNode(newTreeNodeModel);
154             
155             // Save the node
156             this.getTreeNodeModels().put(table, newNode);
157 
158             // Get the parent database
159             databaseNode = this.getTreeNodeModels().get(table.getParent());
160             
161             // Report NullPointerException #1670
162             if (databaseNode != null) {
163                 
164                 // Add the table to the database
165                 treeModel.insertNodeInto(newNode, databaseNode, databaseNode.getChildCount());
166                 
167             } else {
168                 
169                 LOGGER.log(LogLevelUtil.CONSOLE_ERROR, "Missing database for table {}.", () -> table);
170             }
171         }
172 
173         if (databaseNode != null) {
174             
175             // Open the database node
176             this.expandPath(new TreePath(databaseNode.getPath()));
177             
178             // The database has just been search (avoid double check)
179             ((AbstractNodeModel) databaseNode.getUserObject()).setLoaded(true);
180         }
181     }

183     public void createValuesTab(String[][] data, String[] columnNames, AbstractElementDatabase table) {
184 
185         // Report NullPointerException #1683
186         DefaultMutableTreeNode node = this.getTreeNodeModels().get(table);
187         
188         if (node != null) {
189             
190             // Get the node
191             AbstractNodeModel progressingTreeNodeModel = (AbstractNodeModel) node.getUserObject();
192             
193             // Update the progress value of the model, end the progress
194             progressingTreeNodeModel.setIndexProgress(table.getChildCount());
195             
196             // Mark the node model as 'no stop/pause/resume button'
197             progressingTreeNodeModel.setRunning(false);
198             
199             MediatorHelper.tabResults().createValuesTab(data, columnNames, table);
200         }
201     }
