357         
358         var itemSqlEngine = new JMenuItem(I18nUtil.valueByKey(KEY_MENU_SQL_ENGINE));

360         
361         // Render the SQL Engine dialog behind scene

365             
366             for (var i = 0; i < MediatorHelper.tabResults().getTabCount() ; i++) {
367                 if (titleTabSqlEngine.equals(MediatorHelper.tabResults().getTitleAt(i))) {
368                     
369                     MediatorHelper.tabResults().setSelectedIndex(i);

371                 }
372             }
373             
374             CreateTabHelper.initializeSplitOrientation();

376             var panelSqlEngine = new SqlEngine();

378             MediatorHelper.tabResults().addTab(titleTabSqlEngine, panelSqlEngine);

383             // Create a custom tab header with close button

387             // Apply the custom header to the tab
388             MediatorHelper.tabResults().setTabComponentAt(MediatorHelper.tabResults().indexOfComponent(panelSqlEngine), header);

394     private JMenuItem initializeItemPreferences() {
395         
396         JMenuItem itemPreferences = new JMenuItemWithMargin(I18nUtil.valueByKey(KEY_MENU_PREFERENCES), 'P');
397         I18nViewUtil.addComponentForKey(KEY_MENU_PREFERENCES, itemPreferences);
398         
399         // Render the Preferences dialog behind scene
400         var titleTabPreferences = "Preferences";
401         
402         // Single rendering
403         var panelPreferences = new PanelPreferences();
404         
405         itemPreferences.addActionListener(actionEvent -> {
406             
407             for (var i = 0; i < MediatorHelper.tabResults().getTabCount() ; i++) {
408                 if (titleTabPreferences.equals(MediatorHelper.tabResults().getTitleAt(i))) {
409                     
410                     MediatorHelper.tabResults().setSelectedIndex(i);
411                     return;
412                 }
413             }
414             
415             CreateTabHelper.initializeSplitOrientation();
416             
417             AdjustmentListener singleItemScroll = adjustmentEvent -> {
418                 
419                 // The user scrolled the List (using the bar, mouse wheel or something else):
420                 if (adjustmentEvent.getAdjustmentType() == AdjustmentEvent.TRACK) {
421                     
422                     // Jump to the next "block" (which is a row".
423                     adjustmentEvent.getAdjustable().setBlockIncrement(100);
424                     adjustmentEvent.getAdjustable().setUnitIncrement(100);
425                 }
426             };
427 
428             var scroller = new LightScrollPane(1, 0, 0, 0, panelPreferences);
429             scroller.scrollPane.getVerticalScrollBar().addAdjustmentListener(singleItemScroll);
430             
431             MediatorHelper.tabResults().addTab(titleTabPreferences, scroller);
432 
433             // Focus on the new tab
434             MediatorHelper.tabResults().setSelectedComponent(scroller);
435 
436             // Create a custom tab header with close button
437             var header = new TabHeader(I18nViewUtil.valueByKey(KEY_MENU_PREFERENCES), UiUtil.ICON_COG, panelPreferences.getPanelTampering());
438             I18nViewUtil.addComponentForKey(KEY_MENU_PREFERENCES, header.getTabTitleLabel());
439 
440             // Apply the custom header to the tab
441             MediatorHelper.tabResults().setTabComponentAt(MediatorHelper.tabResults().indexOfComponent(scroller), header);
442         });
443         
444         return itemPreferences;
445     }

955     private void switchOrientation(Locale oldLocale, Locale newLocale, boolean isStartup) {
956         
957         var componentOrientation = ComponentOrientation.getOrientation(I18nUtil.getLocaleDefault());
958         MediatorHelper.frame().applyComponentOrientation(componentOrientation);
959         
960         if (!ComponentOrientation.getOrientation(oldLocale).equals(ComponentOrientation.getOrientation(newLocale))) {
961             
962             JSplitPane splitPaneLeftRight = MediatorHelper.frame().getSplitHorizontalTopBottom().getSplitVerticalLeftRight();
963             
964             var componentLeft = splitPaneLeftRight.getLeftComponent();
965             var componentRight = splitPaneLeftRight.getRightComponent();
966 
967             // Reset components
968             splitPaneLeftRight.setLeftComponent(null);
969             splitPaneLeftRight.setRightComponent(null);
970             
971             splitPaneLeftRight.setLeftComponent(componentRight);
972             splitPaneLeftRight.setRightComponent(componentLeft);
973             
974             if (isStartup) {
975                 splitPaneLeftRight.setDividerLocation(
976                     splitPaneLeftRight.getDividerLocation()
977                 );
978             } else {
979                 splitPaneLeftRight.setDividerLocation(
980                     splitPaneLeftRight.getWidth() -
981                     splitPaneLeftRight.getDividerLocation()
982                 );
983             }
984         }
985         
986         MediatorHelper.tabResults().setComponentOrientation(ComponentOrientation.getOrientation(newLocale));
987     }

989     private void switchI18nComponents(Locale newLocale) {
990         
991         for (String key: I18nViewUtil.keys()) {
992             
993             String textI18n = I18nViewUtil.valueByKey(key, newLocale);
994             
995             for (Object componentSwing: I18nViewUtil.componentsByKey(key)) {
996                 if (componentSwing instanceof JTextFieldPlaceholder) {
997                     // Textfield does not need <html> tags for asian fonts
998                     ((JTextFieldPlaceholder) componentSwing).setPlaceholderText(I18nUtil.valueByKey(key));
999                 } else if (componentSwing instanceof JToolTipI18n) {
1000                     ((JToolTipI18n) componentSwing).setText(textI18n);
1001                 } else if (componentSwing instanceof JLabel) {
1002                     ((JLabel) componentSwing).setText(textI18n);
1003                 } else if (componentSwing instanceof JCheckBoxMenuItem) {
1004                     ((JCheckBoxMenuItem) componentSwing).setText(textI18n);
1005                 } else if (componentSwing instanceof JMenuItem) {
1006                     ((JMenuItem) componentSwing).setText(textI18n);
1007                 } else if (componentSwing instanceof JButtonStateful) {
1008                     ((JButtonStateful) componentSwing).setText(textI18n);
1009                 } else if (componentSwing instanceof NodeModelEmpty) {
1010                     ((NodeModelEmpty) componentSwing).setText(textI18n);
1011                 } else {
1012                     ((JTextComponent) componentSwing).setText(textI18n);
1013                 }
1014             }
1015         }
1016     }
