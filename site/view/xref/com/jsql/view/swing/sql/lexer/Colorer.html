31  /**
32   * Run the Syntax Highlighting as a separate thread. Things that need to be
33   * colored are messaged to the thread and put in a list.
34   */
35  class Colorer extends Thread {
36      
37      /**
38       * Log4j logger sent to view.
39       */
40      private static final Logger LOGGER = LogManager.getRootLogger();
41      
42      /**
43       * A simple wrapper representing something that needs to be colored. Placed
44       * into an object so that it can be stored in a Vector.
45       */
46      private static class RecolorEvent {
47          
48          private int position;
49          
50          private int adjustment;
51  
52          public RecolorEvent(int position, int adjustment) {
53              this.position = position;
54              this.adjustment = adjustment;
55          }
56  
57          public int getPosition() {
58              return this.position;
59          }
60  
61          public void setPosition(int position) {
62              this.position = position;
63          }
64  
65          public int getAdjustment() {
66              return this.adjustment;
67          }
68  
69          public void setAdjustment(int adjustment) {
70              this.adjustment = adjustment;
71          }
72      }
73      
74      /**
75       * Stores the document we are coloring. We use a WeakReference
76       * so that the document is eligible for garbage collection when
77       * it is no longer being used. At that point, this thread will
78       * shut down itself.
79       */
80      private final WeakReference<HighlightedDocument> document;
81  
82      /**
83       * Keep a list of places in the file that it is safe to restart the
84       * highlighting. This happens whenever the lexer reports that it has
85       * returned to its initial state. Since this list needs to be sorted and
86       * we need to be able to retrieve ranges from it, it is stored in a
87       * balanced tree.
88       */
89      private final TreeSet<DocPosition> iniPositions = new TreeSet<>(DocPositionComparator.instance);
90  
91      /**
92       * As we go through and remove invalid positions we will also be finding
93       * new valid positions. Since the position list cannot be deleted from
94       * and written to at the same time, we will keep a list of the new
95       * positions and simply add it to the list of positions once all the old
96       * positions have been removed.
97       */
98      private final HashSet<DocPosition> newPositions = new HashSet<>();
99  
100     /**
101      * Vector that stores the communication between the two threads.
102      */
103     private final LinkedList<RecolorEvent> events = new LinkedList<>();
104 
105     /**
106      * When accessing the linked list, we need to create a critical section.
107      * we will synchronize on this object to ensure that we don't get unsafe
108      * thread behavior.
109      */
110     private final Object eventsLock = new Object();
111 
112     /**
113      * The amount of change that has occurred before the place in the
114      * document that we are currently highlighting (lastPosition).
115      */
116     private volatile int change = 0;
117 
118     /**
119      * The last position colored
120      */
121     private volatile int lastPosition = -1;
122 
123     /**
124      * Creates the coloring thread for the given document.
125      * 
126      * @param document The document to be colored.
127      */
128     public Colorer(HighlightedDocument document) {
129         super("ThreadColorer");
130         this.document = new WeakReference<>(document);
131     }
132 
133     /**
134      * Tell the Syntax Highlighting thread to take another look at this
135      * section of the document. It will process this as a FIFO. This method
136      * should be done inside a docLock.
137      */
138     public void color(int position, int adjustment) {
139         
140         // figure out if this adjustment effects the current run.
141         // if it does, then adjust the place in the document
142         // that gets highlighted.
143         if (position < this.lastPosition) {
144             
145             if (this.lastPosition < position - adjustment) {
146                 
147                 this.change -= this.lastPosition - position;
148                 
149             } else {
150                 
151                 this.change += adjustment;
152             }
153         }
154         
155         synchronized (this.eventsLock) {
156             
157             if(!this.events.isEmpty()) {
158                 
159                 // check whether to coalesce with current last element
160                 RecolorEvent curLast = this.events.getLast();
161                 
162                 if(adjustment < 0 && curLast.getAdjustment() < 0) {
163                     
164                     // both are removals
165                     if(position == curLast.getPosition()) {
166                         
167                         curLast.setAdjustment(curLast.getAdjustment() + adjustment);
168                         return;
169                     }
170                     
171                 } else if(adjustment >= 0 && curLast.getAdjustment() >= 0) {
172                     
173                     // both are insertions
174                     if(position == curLast.getPosition() + curLast.getAdjustment()) {
175                         
176                         curLast.setAdjustment(curLast.getAdjustment() + adjustment);
177                         return;
178                         
179                     } else if(curLast.getPosition() == position + adjustment) {
180                         
181                         curLast.setPosition(position);
182                         curLast.setAdjustment(curLast.getAdjustment() + adjustment);
183                         
184                         return;
185                     }
186                 }
187             }
188             
189             this.events.add(new RecolorEvent(position, adjustment));
190             this.eventsLock.notifyAll();
191         }
192     }
193 
194     /**
195      * The colorer runs forever and may sleep for long periods of time. It
196      * should be interrupted every time there is something for it to do.
197      */
198     @Override
199     public void run() {
200         
201         while (this.document.get() != null) {
202             
203             try {
204                 RecolorEvent re = new RecolorEvent(0, 0);
205                 synchronized (this.eventsLock) {
206                     
207                     // get the next event to process - stalling until the
208                     // event becomes available
209                     while(this.events.isEmpty() && this.document.get() != null) {
210                         
211                         // stop waiting after a second in case document
212                         // has been cleared.
213                         this.eventsLock.wait(1000);
214                     }
215                     
216                     if (!this.events.isEmpty()) {
217                         
218                         re = this.events.removeFirst();
219                     }
220                 }
221                 this.processEvent(re.getPosition(), re.getAdjustment());
222                 Thread.sleep(100);
223                 
224             } catch(InterruptedException e) {
225                 
226                 LOGGER.log(LogLevelUtil.IGNORE, e, e);
227                 Thread.currentThread().interrupt();
228             }
229         }
230     }
231     
232     private void processEvent(int position, int adjustment) {
233         
234         HighlightedDocument doc = this.document.get();
235         if(doc == null) {
236             
237             return;
238         }
239         
240         // slurp everything up into local variables in case another
241         // thread changes them during coloring process
242         AttributeSet globalStyle = doc.getGlobalStyle();
243         Lexer syntaxLexer = doc.getSyntaxLexer();
244         DocumentReader documentReader = doc.getDocumentReader();
245         Object docLock = doc.getDocumentLock();
246 
247         if (globalStyle != null) {
248             
249             int start = Math.min(position, position + adjustment);
250             int stop = Math.max(position, position + adjustment);
251             
252             synchronized (docLock) {
253                 
254                 doc.setCharacterAttributes(start, stop - start, globalStyle, true);
255             }
256             
257             return;
258         }
259         
260         SortedSet<DocPosition> workingSet;
261         Iterator<DocPosition> workingIt;
262         DocPosition startRequest = new DocPosition(position);
263         DocPosition endRequest = new DocPosition(position + Math.abs(adjustment));
264         DocPosition dp;
265         DocPosition dpStart = null;
266         DocPosition dpEnd;
267 
268         // find the starting position. We must start at least one
269         // token before the current position
270         try {
271             // all the good positions before
272             workingSet = this.iniPositions.headSet(startRequest);
273             // the last of the stuff before
274             dpStart = workingSet.last();
275             
276         } catch (NoSuchElementException e) {
277             
278             // if there were no good positions before the requested
279             // start,
280             // we can always start at the very beginning.
281             dpStart = new DocPosition(0);
282             
283             LOGGER.log(LogLevelUtil.IGNORE, e);
284         }
285 
286         // if stuff was removed, take any removed positions off the
287         // list.
288         if (adjustment < 0) {
289             
290             workingSet = this.iniPositions.subSet(startRequest, endRequest);
291             workingIt = workingSet.iterator();
292             
293             while (workingIt.hasNext()) {
294                 
295                 workingIt.next();
296                 workingIt.remove();
297             }
298         }
299 
300         // adjust the positions of everything after the
301         // insertion/removal.
302         workingSet = this.iniPositions.tailSet(startRequest);
303         workingIt = workingSet.iterator();
304         while (workingIt.hasNext()) {
305             
306             workingIt.next().adjustPosition(adjustment);
307         }
308 
309         // now go through and highlight as much as needed
310         workingSet = this.iniPositions.tailSet(dpStart);
311         workingIt = workingSet.iterator();
312         dp = null;
313         
314         if (workingIt.hasNext()) {
315             
316             dp = workingIt.next();
317         }
318         
319         try {
320             Token t;
321             boolean done = false;
322             dpEnd = dpStart;
323             
324             synchronized (docLock) {
325                 
326                 // we are playing some games with the lexer for
327                 // efficiency.
328                 // we could just create a new lexer each time here,
329                 // but instead,
330                 // we will just reset it so that it thinks it is
331                 // starting at the
332                 // beginning of the document but reporting a funny
333                 // start position.
334                 // Reseting the lexer causes the close() method on
335                 // the reader
336                 // to be called but because the close() method has
337                 // no effect on the
338                 // DocumentReader, we can do this.
339                 syntaxLexer.reset(documentReader, 0, dpStart
340                         .getPosition(), 0);
341                 // After the lexer has been set up, scroll the
342                 // reader so that it
343                 // is in the correct spot as well.
344                 documentReader.seek(dpStart.getPosition());
345                 // we will highlight tokens until we reach a good
346                 // stopping place.
347                 // the first obvious stopping place is the end of
348                 // the document.
349                 // the lexer will return null at the end of the
350                 // document and wee
351                 // need to stop there.
352                 t = syntaxLexer.getNextToken();
353             }
354             
355             this.newPositions.add(dpStart);
356             
357             while (!done && t != null) {
358                 // this is the actual command that colors the stuff.
359                 // Color stuff with the description of the styles
360                 // stored in tokenStyles.
361                 if (t.getCharEnd() <= doc.getLength()) {
362                     
363                     doc.setCharacterAttributes(
364                         t.getCharBegin() + this.change,
365                         t.getCharEnd() - t.getCharBegin(),
366                         TokenStyles.getStyle(t.getDescription()),
367                         true
368                     );
369                     // record the position of the last bit of
370                     // text that we colored
371                     dpEnd = new DocPosition(t.getCharEnd());
372                 }
373                 
374                 this.lastPosition = t.getCharEnd() + this.change;
375                 
376                 // The other more complicated reason for doing no
377                 // more highlighting
378                 // is that all the colors are the same from here on
379                 // out anyway.
380                 // We can detect this by seeing if the place that
381                 // the lexer returned
382                 // to the initial state last time we highlighted is
383                 // the same as the
384                 // place that returned to the initial state this
385                 // time.
386                 // As long as that place is after the last changed
387                 // text, everything
388                 // from there on is fine already.
389                 if (t.getState() == Token.INITIAL_STATE) {
390                     
391                     // look at all the positions from last time that
392                     // are less than or
393                     // equal to the current position
394                     while (dp != null && dp.getPosition() <= t.getCharEnd()) {
395                         
396                         if (dp.getPosition() == t.getCharEnd() && dp.getPosition() >= endRequest.getPosition()) {
397                             
398                             // we have found a state that is the
399                             // same
400                             done = true;
401                             dp = null;
402                             
403                         } else if (workingIt.hasNext()) {
404                             
405                             // didn't find it, try again.
406                             dp = workingIt.next();
407                             
408                         } else {
409                             
410                             // didn't find it, and there is no more
411                             // info from last
412                             // time. This means that we will just
413                             // continue
414                             // until the end of the document.
415                             dp = null;
416                         }
417                     }
418                     
419                     // so that we can do this check next time,
420                     // record all the
421                     // initial states from this time.
422                     this.newPositions.add(dpEnd);
423                 }
424                 
425                 synchronized (docLock) {
426                     
427                     t = syntaxLexer.getNextToken();
428                 }
429             }
430 
431             // remove all the old initial positions from the place
432             // where
433             // we started doing the highlighting right up through
434             // the last
435             // bit of text we touched.
436             workingIt = this.iniPositions.subSet(dpStart, dpEnd).iterator();
437             while (workingIt.hasNext()) {
438                 
439                 workingIt.next();
440                 workingIt.remove();
441             }
442 
443             // Remove all the positions that are after the end of
444             // the file.:
445             workingIt = this.iniPositions.tailSet(new DocPosition(doc.getLength())).iterator();
446             while (workingIt.hasNext()) {
447                 
448                 workingIt.next();
449                 workingIt.remove();
450             }
451 
452             // and put the new initial positions that we have found
453             // on the list.
454             this.iniPositions.addAll(this.newPositions);
455             this.newPositions.clear();
456             
457         } catch (IOException e) {
458             
459             LOGGER.log(LogLevelUtil.IGNORE, e);
460         }
461         
462         synchronized (docLock) {
463             
464             this.lastPosition = -1;
465             this.change = 0;
466         }
467     }
468     
469     /**
470      * Stop the thread's method run()
471      */
472     public void stopThread() {
473         this.document.clear();
474     }
