761   /**
762    * Refills the input buffer.
763    *
764    * @return      <code>false</code>, iff there was new input.
765    * 
766    * @exception   IOException  if any I/O-Error occurs
767    */
768   private boolean zzRefill() throws IOException {
769 
770     /* first: make room (if you can) */
771     if (this.zzStartRead > 0) {
772       this.zzEndRead += this.zzFinalHighSurrogate;
773       this.zzFinalHighSurrogate = 0;
774       System.arraycopy(this.zzBuffer, this.zzStartRead,
775                        this.zzBuffer, 0,
776                        this.zzEndRead-this.zzStartRead);
777 
778       /* translate stored positions */
779       this.zzEndRead-= this.zzStartRead;
780       this.zzCurrentPos-= this.zzStartRead;
781       this.zzMarkedPos-= this.zzStartRead;
782       this.zzStartRead = 0;
783     }
784 
785     /* is the buffer big enough? */
786     if (this.zzCurrentPos >= this.zzBuffer.length - this.zzFinalHighSurrogate) {
787       /* if not: blow it up */
788       char newBuffer[] = new char[this.zzBuffer.length*2];
789       System.arraycopy(this.zzBuffer, 0, newBuffer, 0, this.zzBuffer.length);
790       this.zzBuffer = newBuffer;
791       this.zzEndRead += this.zzFinalHighSurrogate;
792       this.zzFinalHighSurrogate = 0;
793     }
794 
795     /* fill the buffer with new input */
796     int requested = this.zzBuffer.length - this.zzEndRead;
797     int numRead = this.zzReader.read(this.zzBuffer, this.zzEndRead, requested);
798 
799     /* not supposed to occur according to specification of java.io.Reader */
800     if (numRead == 0) {
801       throw new IOException("Reader returned 0 characters. See JFlex examples for workaround.");
802     }
803     if (numRead > 0) {
804       this.zzEndRead += numRead;
805       /* If numRead == requested, we might have requested to few chars to
806          encode a full Unicode character. We assume that a Reader would
807          otherwise never return half characters. */
808       if (numRead == requested) {
809         if (Character.isHighSurrogate(this.zzBuffer[this.zzEndRead - 1])) {
810           --this.zzEndRead;
811           this.zzFinalHighSurrogate = 1;
812         }
813       }
814       /* potentially more input available */
815       return false;
816     }
817 
818     /* numRead < 0 ==> end of stream */
819     return true;
820   }
