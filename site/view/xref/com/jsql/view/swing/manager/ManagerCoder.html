31  /**
32   * Manager to code/decode string in various methods.
33   */
34  public class ManagerCoder extends JPanel implements Manager {
35      
36      /**
37       * User input to encode.
38       */
39      private final JTextArea textInput;
40  
41      /**
42       * JTextArea displaying result of encoding/decoding.
43       */
44      private final JTextPane result;
45  
46      /**
47       * Encoding choice by user.
48       */
49      private JMenuItem menuMethod;
50  
51      private final transient CoderListener actionCoder = new CoderListener(this);
52      
53      private class ChangeMenuListener implements ChangeListener {
54          
55          private final String nameMethod;
56          
57          ChangeMenuListener(String nameMethod) {
58              this.nameMethod = nameMethod;
59          }
60          
61          @Override
62          public void stateChanged(ChangeEvent e) {
63              if (e.getSource() instanceof JMenuItem) {
64                  
65                  JMenuItem item = (JMenuItem) e.getSource();
66                  
67                  if (item.isSelected() || item.isArmed()) {
68                      ManagerCoder.this.actionCoder.actionPerformed(this.nameMethod);
69                  }
70              }
71          }
72      }
73  
74      /**
75       * Create a panel to encode a string.
76       */
77      public ManagerCoder() {
78          
79          super(new BorderLayout());
80  
81          this.textInput = new JPopupTextArea(new JTextAreaPlaceholder("Type a string to convert")).getProxy();
82          this.textInput.getCaret().setBlinkRate(500);
83          this.textInput.setEditable(true);
84          this.textInput.setLineWrap(true);
85          this.textInput.setName("textInputManagerCoder");
86          
87          this.textInput.getDocument().addDocumentListener(new DocumentListenerEditing() {
88              @Override
89              public void process() {
90                  ManagerCoder.this.actionCoder.actionPerformed();
91              }
92          });
93  
94          JPanel topMixed = this.initializeTopPanel();
95  
96          this.result = new JPopupTextPane("Result of conversion").getProxy();
97          this.result.setContentType("text/html");
98          this.result.setEditorKit(new HTMLEditorKitTextPaneWrap());
99          this.result.setName("resultManagerCoder");
100         
101         var bottom = new JPanel(new BorderLayout());
102         bottom.add(new LightScrollPane(0, 0, 0, 0, this.result), BorderLayout.CENTER);
103 
104         var divider = new JSplitPaneWithZeroSizeDivider(JSplitPane.VERTICAL_SPLIT);
105         divider.setBorder(BorderFactory.createEmptyBorder(0, 0, 0, 0));
106         divider.setDividerSize(0);
107         divider.setResizeWeight(0.5);
108 
109         divider.setTopComponent(topMixed);
110         divider.setBottomComponent(bottom);
111 
112         this.add(divider, BorderLayout.CENTER);
113     }
114 
115     private JPanel initializeTopPanel() {
116         
117         var topMixed = new JPanel(new BorderLayout());
118 
119         final var middleLine = new JPanel();
120         middleLine.setLayout(new BorderLayout());
121         middleLine.setBorder(BorderFactory.createEmptyBorder(1, 0, 1, 1));
122 
123         var comboMenubar = this.initializeMenuBarCoder();
124         
125         this.menuMethod.setText("Encode to Base64");
126         
127         middleLine.add(comboMenubar);
128 
129         topMixed.add(new LightScrollPane(0, 0, 1, 0, this.textInput), BorderLayout.CENTER);
130         topMixed.add(middleLine, BorderLayout.SOUTH);
131         
132         return topMixed;
133     }
134 
135     private MenuBarCoder initializeMenuBarCoder() {
136         
137         Map<String, JMenu> mapMenus = new LinkedHashMap<>();
138         
139         mapMenus.put("Base16", new JMenu("Base16"));
140         mapMenus.put("Base32", new JMenu("Base32"));
141         mapMenus.put("Base58", new JMenu("Base58"));
142         mapMenus.put("Base64", new JMenu("Base64"));
143         mapMenus.put("Hex", new JMenu("Hex"));
144         mapMenus.put("Url", new JMenu("Url"));
145         mapMenus.put("Unicode", new JMenu("Unicode"));
146         
147         var menuHtml = new JMenu("Html");
148         mapMenus.put("Html", menuHtml);
149         mapMenus.put("Base64(zipped)", new JMenu("Base64(zipped)"));
150         mapMenus.put("Hex(zipped)", new JMenu("Hex(zipped)"));
151 
152         var menuEncodeHtmlDecimal = new JMenuItem("Encode to Html (decimal)");
153         menuHtml.add(menuEncodeHtmlDecimal);
154         menuEncodeHtmlDecimal.addActionListener(this.actionCoder);
155         menuEncodeHtmlDecimal.addChangeListener(new ChangeMenuListener("Encode to Html (decimal)"));
156         
157         mapMenus
158         .forEach((key, value) -> {
159 
160             var menuEncode = new JMenuItem("Encode to " + key);
161             menuEncode.addActionListener(this.actionCoder);
162             menuEncode.addChangeListener(new ChangeMenuListener("Encode to " + key));
163             menuEncode.setName("encodeTo" + key);
164 
165             var menuDecode = new JMenuItem("Decode from " + key);
166             menuDecode.addActionListener(this.actionCoder);
167             menuDecode.addChangeListener(new ChangeMenuListener("Decode from " + key));
168             menuDecode.setName("decodeFrom" + key);
169 
170             value.add(menuEncode);
171             value.add(menuDecode);
172             value.setName(key);
173         });
174 
175         mapMenus.put("Hash", new JMenu("Hash"));
176         mapMenus.get("Hash").setName("Hash");
177         
178         Stream.of("Adler32", "Crc16", "Crc32", "Crc64", "Md2", "Md4", "Md5", "Sha-1", "Sha-256", "Sha-384", "Sha-512", "Mysql")
179         .forEach(hash -> {
180             
181             var menuEncode = new JMenuItem("Hash to "+ hash);
182             menuEncode.addActionListener(this.actionCoder);
183             menuEncode.addChangeListener(new ChangeMenuListener("Hash to "+ hash));
184             menuEncode.setName("hashTo"+ hash);
185             
186             mapMenus.get("Hash").add(menuEncode);
187         });
188 
189         JMenu comboMenu = MenuBarCoder.createMenu("Choose method...");
190         this.menuMethod = comboMenu;
191         this.menuMethod.setName("menuMethodManagerCoder");
192         
193         for (JMenu menu: mapMenus.values()) {
194             comboMenu.add(menu);
195         }
196 
197         var comboMenubar = new MenuBarCoder(comboMenu);
198         comboMenubar.setOpaque(false);
199         comboMenubar.setBorder(null);
200         
201         return comboMenubar;
202     }
